  'https://clario-complete-backend-y5cd.onrender.com', 
  'http://localhost:8080', 
  'http://localhost:5173'

  credentials: true,
  methods: ['GET', 'POST', 'PUT', 'DELETE', 'PATCH', 'OPTIONS'],
  allowedHeaders: ['Content-Type', 'Authorization', 'X-Requested-With']
}));
